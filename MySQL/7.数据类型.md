# 整型

| 类型        | 字节 |
| ----------- | ---- |
| tinyint     | 1    |
| smallint    | 2    |
| mediumint   | 3    |
| int/integer | 4    |
| bigint      | 8    |

~~~mysql
CREATE TABLE IF NOT EXISTS inttest(
i1 INT,
i2 INT(5),
i3 INT(5) ZEROFILL, #用0填充。
i4 INT UNSIGNED  #设置为无符号
);

INSERT INTO inttest
VALUES(123456,123456,123456,123456);

INSERT INTO inttest
VALUES(123,123,123,123456);
~~~
